# Station Improvement Hub
station_improvement_hub_0 = {
	entity = "construction_platform_entity"
	construction_entity = "construction_platform_entity"
	portrait = "GFX_megastructure_construction_background"
	place_entity_on_planet_plane = yes
	entity_offset = { x = -27 y = -27 }
	build_time = 720
	use_planet_resource = no
	resources = {
		category = megastructures
		cost = {
			alloys = 2500
			unity = 1000
		}

		upkeep = {
			energy = 5
		}
	}

	prerequisites = { "stellaris_incremental_tech_station_oversight" }
	#show_prereqs = yes

	possible = {
		exists = starbase
		custom_tooltip = {
			fail_text = "requires_inside_border"
			is_inside_border = from
		}
		custom_tooltip = {
			fail_text = "requires_surveyed_system"
			NOT = {
				any_system_planet = {
					is_surveyed = {
						who = prev.from
						status = no
					}
				}
			}
		}
	}

	placement_rules = {
		planet_possible = {
			custom_tooltip = {
				fail_text = "requires_survey_not_habitable"
				is_surveyed = {			# prevent leaking habitability information
					who = prev.from
					status = yes
				}
				colonizable_planet = no
			}
			custom_tooltip = {
				fail_text = "requires_no_anomaly"
				NOT = { has_anomaly = yes }
			}
			custom_tooltip = {
				fail_text = "requires_not_minor_planetary_body"
				NOR = {
					is_asteroid = yes
					is_moon = yes
				}
			}
			NOR = {
				has_planet_flag = megastructure
				has_planet_flag = has_megastructure
				solar_system = {
					has_star_flag = ring_world_built
				}
				is_planet_class = pc_ringworld_habitable
				is_planet_class = pc_ringworld_habitable_damaged
				is_planet_class = pc_ringworld_tech
				is_planet_class = pc_ringworld_tech_damaged
				is_planet_class = pc_ringworld_seam
				is_planet_class = pc_ringworld_seam_damaged
				is_planet_class = pc_habitat
			}
			custom_tooltip = {
				fail_text = "requires_not_star"
				is_star = no
			}
			custom_tooltip = {
				fail_text = "requires_not_ring_world"
				is_ringworld = no
			}
		}
	}

	# root = system
	# from = country
	ai_weight = {
		factor = 15

		modifier = {
			factor = 0.1
			starbase = { NOT = { has_starbase_size >= starbase_starfortress } }
		}

		modifier = {
			factor = 0.1
			any_neighbor_system = {
				exists = owner
				NOT = {
					owner = { is_same_value = from }
				}
			}
		}
	}

	on_build_start = {}
	on_build_cancel = {}
	on_build_complete = {
		fromfrom.planet = {
			set_planet_flag = has_megastructure

			if = {
				limit = { has_orbital_station = yes }
				orbital_station = {
					dismantle = yes
				}
			}
		}
	}
}

#ONE TIME
station_improvement_hub_survey_one = {
	entity = ""
	portrait = "GFX_megastructure_construction_background"
	place_entity_on_planet_plane = no
	entity_offset = { x = -27 y = -27 }
	build_time = 600
	resources = {
		category = megastructures
		cost = {
			energy = 5000
			station_data_feed = 500
		}
	}

	upgrade_from = {
		station_improvement_hub_6
	}

	construction_blocks_and_blocked_by = none
	prerequisites = { stellaris_incremental_tech_station_oversight_6 }
	#show_prereqs = yes

	on_build_start = {
		owner = {
			country_event = { id = stellaris_incremental.2 }
		}
	}
	on_build_cancel = {
		owner = {
			country_event = { id = stellaris_incremental.3 }
		}
	}
	on_build_complete = {
		fromfrom.planet = {
			save_global_event_target_as = stellaris_incremental_mega_survey_planet
			solar_system = {
				spawn_megastructure = {
					type = station_improvement_hub_6
					planet = event_target:stellaris_incremental_mega_survey_planet
					owner = space_owner
				}
			}
		}
		owner = {
			country_event = { id = stellaris_incremental.1 }
			country_event = { id = stellaris_incremental.3 }
		}
		
		remove_megastructure = fromfrom

		
	}

}
